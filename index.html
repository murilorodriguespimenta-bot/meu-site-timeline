const items = data.map(e => { const cat = (e.categoria || "").trim(); const cor = mapa[cat] || "#999"; const titulo = e.titulo || "Sem título"; const inicio = e.inicio || ""; const fim = e.fim || ""; const notas = e.notas || ""; // 🔥 FORMATA DATA SEGURA function formatarDataBR(data){ if(!data) return "—"; const d = new Date(data); if(isNaN(d)) return "—"; return `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}/${d.getFullYear()}`; } // 🔥 CALCULAR DURAÇÃO function calcularDuracao(inicio, fim){ const start = new Date(inicio); const end = fim ? new Date(fim) : new Date(inicio); if (isNaN(start) || isNaN(end)) return "—"; let anos = end.getFullYear() - start.getFullYear(); let meses = end.getMonth() - start.getMonth(); let dias = end.getDate() - start.getDate(); if (dias < 0) { meses--; dias += 30; } if (meses < 0) { anos--; meses += 12; } let partes = []; if (anos > 0) partes.push(`${anos} ano${anos>1?'s':''}`); if (meses > 0) partes.push(`${meses} mês${meses>1?'es':''}`); if (dias > 0) partes.push(`${dias} dia${dias>1?'s':''}`); return partes.length ? partes.join(", ") : "Mesmo dia"; } return { id: Number(e.id), content: "", start: inicio, end: fim || null, group: cat, style: `background:${cor};border-color:${cor}`, // 🔥 TOOLTIP GARANTIDO title: ` ${titulo}
Início: ${formatarDataBR(inicio)}
Fim: ${formatarDataBR(fim)}
Duração: ${calcularDuracao(inicio, fim)} ${notas ? `
${notas.substring(0,150)}` : ""} ` }; });
